// https://leetcode.com/problems/min-stack/

class MinStack {
private:
    vector<int> stack;
    vector<int> minStack;

public:
    MinStack() {
    }

    void push(int val) {
        stack.push_back(val);
        // Push the smaller value between current and last min
        if (minStack.empty() || val <= minStack.back()) {
            minStack.push_back(val);
        }
    }

    void pop() {
        if (stack.back() == minStack.back()) {
            minStack.pop_back();
        }
        stack.pop_back();
    }

    int top() {
        return stack.back();
    }

    int getMin() {
        return minStack.back();
    }
};
