class Solution {
public:
    bool isSameTree(TreeNode* p, TreeNode* q) {
        if(!p) return !q;
        if(!q) return !p;
        return (p->val == q->val) && isSameTree(p->left, q->left) && isSameTree(p->right, q->right);
    }
};
/*
TC-> O(min(n,m))
SC-> Recursion Stack: O(min(n,m))ASC
where n and m are the nodes in the tree p and q respectively.
*/
