class Solution {
public:
    int jump(vector<int>& nums) {
        
        int steps = 0;
        int l = 0, r = 0;
        while(r<nums.size()-1){
            int farthest = 0;
            for(int i=l; i<=r; i++){
                farthest = max(farthest, i+nums[i]);
            }
            l = r + 1;
            r = farthest;
            steps += 1;
        }
        return steps;
    }
};
/*
TC-> O(N)
SC-> O(1)
*/
