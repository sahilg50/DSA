// Approach 1: GCD
class Solution {
public:
    string gcdOfStrings(const string &str1, const string &str2) {

        if (str1 + str2 != str2 + str1) return "";
        
        return str1.substr(0, gcd(str1.size(), str2.size()));
    }
};
/*
if N = size(str1) + size(str2)
TC-> O(N^2), because '+' operator on strings
SC-> O(N)
*/
