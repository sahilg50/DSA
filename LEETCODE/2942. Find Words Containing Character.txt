// Approach 1: Iteration 
class Solution {
public:
    vector<int> findWordsContaining(vector<string>& words, char x) {

        vector<int> result;

        for(int ind = 0; ind < words.size(); ind++) {
            for(auto ch : words[ind]) {
                if(ch == x) {
                    result.push_back(ind);

};
